SELECT imie, nazwisko FROM kurierzy INNER JOIN pojazdy ON kurierzy.id_pojazdu = pojazdy.id_pojazdu WHERE rejestracja='L-422Q';
SELECT odbiorcy.imie, odbiorcy.nazwisko FROM odbiorcy INNER JOIN paczki ON paczki.id_odbiorcy = odbiorcy.id_odbiorcy INNER JOIN nadawcy ON nadawcy.id_nadawcy = paczki.id_nadawcy WHERE nadawcy.imie = 'Tatiana';

SELECT COUNT(*) AS 'Liczba Paczek' FROM paczki;
SELECT COUNT(*) AS 'Imie na litere R' FROM kurierzy WHERE imie LIKE 'R%';
